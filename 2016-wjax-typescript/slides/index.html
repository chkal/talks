Why another JavaScript dialect?

Dynamic typing is wired


"3" + 1
// > "31"
            
"3" - 1
// > 2

JavaScript


let n = 3;

TypeScript


let n: number = 3;

Types are great


let n: number = 1;
// > 1
						
n = 2;
// > 2

n = "4";
// Type 'string' is not assignable to type 'number'.

IDE support

Basic types


// numbers
let n: number = 42;

// strings
let s: string = "Foobar";

// booleans
let b: boolean = true;

// arrays
let a: number[] = [ 1, 2, 4, 8 ];

Tuple


let price: [ number, string ];

price = [ 12.99, "EUR" ];
// > OK
	
price = [ "EUR", 12.99 ];
// > Type '[string, number]' is not assignable 
//   to type '[number, string]'.

Enum


enum Currency {
  EUR, USD, JPY, GBP 
};
						
let c: Currency = Currency.EUR;
						
c = "FOOBAR";
// > Property 'FOOBAR' does not exist on
//   type 'typeof Currency'.

Any


let a: any;
						
a = "Foobar";
						
a = false;
						
a = [ 42, "Foobar", true ];
						
a = document.getElementById( "foobar" );

Functions


function formatEuro( value: number ): string {
	return value.toFixed(2) + " €";
}
						
formatEuro( 42 );
// > "42.00 €"
